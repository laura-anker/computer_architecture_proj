; =========================
; JUMP TABLE (LOW MEMORY)
; =========================
JT_READ20:     DATA READ20_LOOP
JT_PRINT20:    DATA PRINT20_LOOP
JT_FIND:       DATA FIND_LOOP
JT_READINT:    DATA READINT
JT_PRINTINT:   DATA PRINTINT
JT_ABS:        DATA ABS
JT_MULT10:     DATA MULT10
JT_READCHAR:   DATA READCHAR
JT_NO_UPDATE:  DATA NO_UPDATE
JT_READINT_LOOP:    DATA READINT_LOOP
JT_READINT_DONE:    DATA READINT_DONE
JT_SET_NEG:         DATA SET_NEG
JT_CHECK_DIGIT:     DATA CHECK_DIGIT
JT_PI_DIV_LOOP:     DATA PI_DIV_LOOP
JT_PI_POS:          DATA PI_POS
JT_PI_CONVERT:      DATA PI_CONVERT
JT_PI_DONE:         DATA PI_DONE
JT_PI_PRINT_LOOP:   DATA PI_PRINT_LOOP
JT_ABS_RET:         DATA ABS_RET
JT_DO_PRINT:        DATA DO_PRINT
JT_AFTER_STORE:     DATA AFTER_STORE
JT_STORE0:          DATA STORE0
JT_STORE1:          DATA STORE1
JT_STORE2:          DATA STORE2
JT_STORE3:          DATA STORE3
JT_STORE4:          DATA STORE4
JT_STORE5:          DATA STORE5
JT_PRINT0:          DATA PRINT0
JT_PRINT1:          DATA PRINT1
JT_PRINT2:          DATA PRINT2
JT_PRINT3:          DATA PRINT3
JT_PRINT4:          DATA PRINT4
;JT_PRINT5:          DATA PRINT5


; ============================================================
; Program 1 — Read 20 signed integers, print them, read a user
; value, find the array element closest to the user value, print
; the user value and the closest value. Handles negatives and
; assembles integers from digit characters (range -32768..32767).
; ============================================================

COUNT:      DATA    20
ARRPTR:     DATA    ARR0
USERVAL:    DATA    0
BESTVAL:    DATA    0
BESTDIFF:   DATA    32767
TEMP:       DATA    0
SIGN:       DATA    0
CHAR:       DATA    0
VALUE:      DATA    0
DIGIT:      DATA    0
TEN:        DATA    10
NEGONE:     DATA    -1

ASCII_ZERO: DATA    48
ASCII_NINE: DATA    57
ASCII_MINUS: DATA   45
ASCII_NL:   DATA    10

; 20-element array
ARR0:       DATA    0
ARR1:       DATA    0
ARR2:       DATA    0
ARR3:       DATA    0
ARR4:       DATA    0
ARR5:       DATA    0
ARR6:       DATA    0
ARR7:       DATA    0
ARR8:       DATA    0
ARR9:       DATA    0
ARR10:      DATA    0
ARR11:      DATA    0
ARR12:      DATA    0
ARR13:      DATA    0
ARR14:      DATA    0
ARR15:      DATA    0
ARR16:      DATA    0
ARR17:      DATA    0
ARR18:      DATA    0
ARR19:      DATA    0

; buffers for printing digits
DIGBUF0:    DATA 0
DIGBUF1:    DATA 0
DIGBUF2:    DATA 0
DIGBUF3:    DATA 0
DIGBUF4:    DATA 0
DIGBUF5:    DATA 0

; ============================================================
; MAIN
; ============================================================

        LOC     40

START:
        ; R2 = COUNT, R1 = pointer to next array slot
        LDR     2,0,COUNT
        LDX     1,ARRPTR

; Read 20 integers into array (assembled from digit characters)
READ20_LOOP:
        JSR     3,JT_READINT,1      ; returns integer in R0
        STR     0,1,0          ; store into *ARRPTR

        LDA     3,1,0          ; bump ARRPTR via R3 (scratch)
        AIR     3,1
        STX     3,ARRPTR
        LDX     1,ARRPTR

        SOB     2,0,JT_READ20,1

; Print the 20 integers (one per line)
        LDR     2,0,COUNT
        LDA     3,0,ARR0
        STX     3,ARRPTR
        LDX     1,ARRPTR

PRINT20_LOOP:
        LDR     0,1,0
        JSR     3,JT_PRINTINT,1

        LDA     3,1,0
        AIR     3,1
        STX     3,ARRPTR
        LDX     1,ARRPTR

        SOB     2,0,JT_PRINT20,1

; Read user search value
        JSR     3,JT_READINT,1
        STR     0,0,USERVAL

; Find closest — iterate array, compute abs(user - element)
        LDR     0,0,COUNT
        STR     0,0,TEMP       ; TEMP = remaining count

        LDA     2,0,ARR0
        STX     2,ARRPTR
        LDX     1,ARRPTR

FIND_LOOP:
        LDR     1,1,0         ; R1 = current array value
        LDR     0,0,USERVAL
        SMR     0,1,0         ; R0 = user - current

        JSR     3,JT_ABS,1        ; R0 = abs difference
        STR     0,0,VALUE

        LDR     2,0,VALUE
        SMR     2,0,BESTDIFF   ; R2 = VALUE - BESTDIFF
        JGE     2,0,JT_NO_UPDATE,1  ; if VALUE >= BESTDIFF, skip
        ; else VALUE < BESTDIFF → update

        LDR     0,0,VALUE
        STR     0,0,BESTDIFF

        LDR     0,1,0
        STR     0,0,BESTVAL

NO_UPDATE:
        LDA     2,1,0
        AIR     2,1
        STX     2,ARRPTR
        LDX     1,ARRPTR

        LDR     0,0,TEMP
        SIR     0,1
        STR     0,0,TEMP

        JGE     0,0,JT_FIND,1

; Print user value and closest value
        LDR     0,0,USERVAL
        JSR     3,JT_PRINTINT,1

        LDR     0,0,BESTVAL
        JSR     3,JT_PRINTINT,1

        HLT

; ============================================================
; SUBROUTINES
; ============================================================

READCHAR:
        IN      1,0
        RFS     0

ABS:
        ; absolute value of R0
        JGE     0,0,JT_ABS_RET
        LDR     1,0,NEGONE
        MLT     0,1
ABS_RET:
        RFS     0

MULT10:
        LDR     1,0,TEN
        MLT     0,1
        RFS     0

PRINTNEWLINE:
        LDR     1,0,ASCII_NL
        OUT     1,0
        RFS     0

; READINT — read signed integer into R0
; Assembles digits until newline; supports optional leading '-'
READINT:
        STR     3,0,TEMP        ; save return address in TEMP

        LDA     0,0,0
        STR     0,0,VALUE

        LDA     0,0,0
        AIR     0,1
        STR     0,0,SIGN

READINT_LOOP:
        JSR     3,JT_READCHAR,1
        STR     1,0,CHAR

        LDR     2,0,CHAR
        SMR     2,0,ASCII_NL
        JZ      2,0,JT_READINT_DONE,1

        LDR     2,0,CHAR
        SMR     2,0,ASCII_MINUS
        JZ      2,0,JT_SET_NEG,1

        LDR     2,0,CHAR
        SMR     2,0,ASCII_ZERO
        JGE     2,0,JT_CHECK_DIGIT,1
        JMA     JT_READINT_LOOP,1

CHECK_DIGIT:
        LDR     2,0,CHAR
        SMR     2,0,ASCII_NINE
        JGE     2,0,JT_READINT_LOOP,1

        LDR     0,0,CHAR
        SMR     0,0,ASCII_ZERO
        STR     0,0,DIGIT

        LDR     0,0,VALUE
        JSR     3,JT_MULT10,1
        STR     0,0,VALUE

        LDR     0,0,VALUE
        AMR     0,0,DIGIT
        STR     0,0,VALUE

        JMA     JT_READINT_LOOP,1

SET_NEG:
        LDR     0,0,NEGONE
        STR     0,0,SIGN
        JMA     JT_READINT_LOOP,1

READINT_DONE:
        LDR     0,0,VALUE
        LDR     1,0,SIGN
        MLT     0,1

        LDR     3,0,TEMP
        RFS     0

; PRINTINT — print integer in R0
PRINTINT:
        STR     3,0,TEMP        ; save return address

        ; handle negative
        JGE     0,0,JT_PI_POS,1

        LDR     1,0,ASCII_MINUS
        OUT     1,0

        LDR     1,0,NEGONE
        MLT     0,1

PI_POS:
        ; if R0 == 0, print '0'
        JNE     0,0,JT_PI_CONVERT,1
        LDR     1,0,ASCII_ZERO
        OUT     1,0
        JMA     JT_PI_DONE,1

PI_CONVERT:
        LDA     1,0,0
        STR     1,0,TEMP        ; TEMP = digit count

PI_DIV_LOOP:
        LDR     1,0,TEN
        DVD     0,1             ; quotient in R0, remainder in R1
        STR     1,0,DIGIT       ; remainder

        ; convert digit to ASCII
        LDR     1,0,DIGIT
        AMR     1,0,ASCII_ZERO

        ; store into DIGBUF[TEMP]
        LDR     2,0,TEMP
        JZ      2,0,JT_STORE0
        SIR     2,1
        JZ      2,0,JT_STORE1
        SIR     2,1
        JZ      2,0,JT_STORE2
        SIR     2,1
        JZ      2,0,JT_STORE3
        SIR     2,1
        JZ      2,0,JT_STORE4
        JMA     JT_STORE5

STORE0: STR 1,0,DIGBUF0
        JMA JT_AFTER_STORE
STORE1: STR 1,0,DIGBUF1
        JMA JT_AFTER_STORE
STORE2: STR 1,0,DIGBUF2
        JMA JT_AFTER_STORE
STORE3: STR 1,0,DIGBUF3
        JMA JT_AFTER_STORE
STORE4: STR 1,0,DIGBUF4
        JMA JT_AFTER_STORE
STORE5: STR 1,0,DIGBUF5

AFTER_STORE:
        ; TEMP++
        LDR     1,0,TEMP
        AIR     1,1
        STR     1,0,TEMP

        ; continue if quotient > 0
        JNE     0,0,JT_PI_DIV_LOOP,1

PI_PRINT_LOOP:
        LDR     1,0,TEMP
        SIR     1,1
        STR     1,0,TEMP

        LDR     2,0,TEMP
        JZ      2,0,JT_PRINT0
        SIR     2,1
        JZ      2,0,JT_PRINT1
        SIR     2,1
        JZ      2,0,JT_PRINT2
        SIR     2,1
        JZ      2,0,JT_PRINT3
        SIR     2,1
        JZ      2,0,JT_PRINT4
        JMA     PRINT5

PRINT0: LDR 1,0,DIGBUF0
        JMA JT_DO_PRINT
PRINT1: LDR 1,0,DIGBUF1
        JMA JT_DO_PRINT
PRINT2: LDR 1,0,DIGBUF2
        JMA JT_DO_PRINT
PRINT3: LDR 1,0,DIGBUF3
        JMA JT_DO_PRINT
PRINT4: LDR 1,0,DIGBUF4
        JMA JT_DO_PRINT
PRINT5: LDR 1,0,DIGBUF5

DO_PRINT:
        OUT     1,0

        ; if TEMP != 0, keep printing; else we’re done
        LDR     1,0,TEMP
        JNE     1,0,JT_PI_PRINT_LOOP,1
        JMA     JT_PI_DONE,1

PI_DONE:
        LDR     1,0,ASCII_NL
        OUT     1,0

        LDR     3,0,TEMP
        RFS     0
