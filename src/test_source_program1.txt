;-------------------------------------
; READ 20 NUMBERS
;-------------------------------------

        LDX 1,12       ; X1 = address of ARR0
        LDX 2,6        ; X2 = COUNT = 20

READLOOP:
        IN 1,0         ; read number into R1
        STX 1,1,0      ; store R1 at memory[X1]

        LDA 3,0,1      ; R3 = 1
        AMR 1,3        ; X1 = X1 + 1

        SOB 2,0,READLOOP   ; loop 20 times

;-------------------------------------
; PRINT NUMBERS
;-------------------------------------

        LDX 1,12       ; reset X1 to ARR0
        LDX 2,6        ; reset loop counter

PRINTLOOP:
        LDX 3,1,0      ; R3 = memory[X1]
        OUT 3,0        ; print it

        LDA 4,0,1      ; R4 = 1
        AMR 1,4        ; X1 = X1 + 1

        SOB 2,0,PRINTLOOP

;-------------------------------------
; GET USER NUMBER
;-------------------------------------

        IN 2,0         ; user number -> R2
        STR 2,0,9      ; store USER

;-------------------------------------
; FIND CLOSEST
;-------------------------------------

        LDX 1,12       ; X1 = ARR0
        LDX 2,6        ; X2 = COUNT = 20

FINDLOOP:
        LDX 3,1,0      ; R3 = ARR[index]

        LDR 4,0,9      ; R4 = USER
        SMR 3,4        ; R3 = ARR[i] - USER

        ; absolute value
        JGE 3,0,POS
        NOT 3
        AIR 3,1
POS:

        LDR 5,0,10     ; R5 = MIN
        SMR 5,3        ; R5 = MIN - diff
        JGE 5,0,SKIP   ; if MIN <= diff, skip update

        STR 3,0,10     ; MIN = diff
        STR 3,0,11     ; CLOSE = diff

SKIP:
        LDA 6,0,1      ; R6 = 1
        AMR 1,6        ; X1 = X1 + 1

        SOB 2,0,FINDLOOP

;-------------------------------------
; PRINT RESULTS
;-------------------------------------

        LDR 2,0,9      ; USER
        OUT 2,0

        LDR 3,0,11     ; CLOSEST
        OUT 3,0

        HLT
